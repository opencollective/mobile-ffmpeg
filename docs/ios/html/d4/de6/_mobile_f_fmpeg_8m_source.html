96 + (int)executeWithArguments: (NSArray*)arguments {
97  lastCommandOutput = [[NSMutableString alloc] init];
98 
99  char **commandCharPArray = (char **)av_malloc(sizeof(char*) * ([arguments count] + 1));
100 
101  /* PRESERVING CALLING FORMAT
102  *
103  * ffmpeg <arguments>
104  */
105  commandCharPArray[0] = (char *)av_malloc(sizeof(char) * ([LIB_NAME length] + 1));
106  strcpy(commandCharPArray[0], [LIB_NAME UTF8String]);
107 
108  for (int i=0; i < [arguments count]; i++) {
109  NSString *argument = [arguments objectAtIndex:i];
110  commandCharPArray[i + 1] = (char *) [argument UTF8String];
111  }
112 
113  // RUN
114  if ([MobileFFmpeg compareAndSet:FALSE with:TRUE]) {
115  lastReturnCode = execute(([arguments count] + 1), commandCharPArray);
116  [MobileFFmpeg compareAndSet:TRUE with:FALSE];
117  } else {
118  NSLog(@"execute cancelled. Multiple executions not supported.");
119  lastReturnCode = RETURN_CODE_MULTIPLE_EXECUTIONS_NOT_ALLOWED;
120  }
121 
122  // CLEANUP
123  av_free(commandCharPArray[0]);
124  av_free(commandCharPArray);
125 
126  return lastReturnCode;
127 }

232 + (NSArray*)parseArguments: (NSString*)command {
233  NSMutableArray *argumentArray = [[NSMutableArray alloc] init];
234  NSMutableString *currentArgument = [[NSMutableString alloc] init];
235 
236  bool singleQuoteStarted = false;
237  bool doubleQuoteStarted = false;
238 
239  for (int i = 0; i < command.length; i++) {
240  unichar previousChar;
241  if (i > 0) {
242  previousChar = [command characterAtIndex:(i - 1)];
243  } else {
244  previousChar = 0;
245  }
246  char currentChar = [command characterAtIndex:i];
247 
248  if (currentChar == ' ') {
249  if (singleQuoteStarted || doubleQuoteStarted) {
250  [currentArgument appendFormat: @"%c", currentChar];
251  } else if ([currentArgument length] > 0) {
252  [argumentArray addObject: currentArgument];
253  currentArgument = [[NSMutableString alloc] init];
254  }
255  } else if (currentChar == '\'' && (previousChar == 0 || previousChar != '\\')) {
256  if (singleQuoteStarted) {
257  singleQuoteStarted = false;
258  } else if (doubleQuoteStarted) {
259  [currentArgument appendFormat: @"%c", currentChar];
260  } else {
261  singleQuoteStarted = true;
262  }
263  } else if (currentChar == '\"' && (previousChar == 0 || previousChar != '\\')) {
264  if (doubleQuoteStarted) {
265  doubleQuoteStarted = false;
266  } else if (singleQuoteStarted) {
267  [currentArgument appendFormat: @"%c", currentChar];
268  } else {
269  doubleQuoteStarted = true;
270  }
271  } else {
272  [currentArgument appendFormat: @"%c", currentChar];
273  }
274  }
275 
276  if ([currentArgument length] > 0) {
277  [argumentArray addObject: currentArgument];
278  }
279 
280  return argumentArray;
281 }
