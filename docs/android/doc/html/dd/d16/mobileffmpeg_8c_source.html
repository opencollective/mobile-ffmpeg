149 static void avutil_log_format_line(void *avcl, int level, const char *fmt, va_list vl, AVBPrint part[4], int *print_prefix) {
150  int flags = av_log_get_flags();
151  AVClass* avc = avcl ? *(AVClass **) avcl : NULL;
152  av_bprint_init(part+0, 0, 1);
153  av_bprint_init(part+1, 0, 1);
154  av_bprint_init(part+2, 0, 1);
155  av_bprint_init(part+3, 0, 65536);
156 
157  if (*print_prefix && avc) {
158  if (avc->parent_log_context_offset) {
159  AVClass** parent = *(AVClass ***) (((uint8_t *) avcl) +
160  avc->parent_log_context_offset);
161  if (parent && *parent) {
162  av_bprintf(part+0, "[%s @ %p] ",
163  (*parent)->item_name(parent), parent);
164  }
165  }
166  av_bprintf(part+1, "[%s @ %p] ",
167  avc->item_name(avcl), avcl);
168  }
169 
170  if (*print_prefix && (level > AV_LOG_QUIET) && (flags & AV_LOG_PRINT_LEVEL))
171  av_bprintf(part+2, "[%s] ", avutil_log_get_level_str(level));
172 
173  av_vbprintf(part+3, fmt, vl);
174 
175  if(*part[0].str || *part[1].str || *part[2].str || *part[3].str) {
176  char lastc = part[3].len && part[3].len <= part[3].size ? part[3].str[part[3].len - 1] : 0;
177  *print_prefix = lastc == '\n' || lastc == '\r';
178  }
179 }

265 void appendLastCommandOutput(const char *logMessage) {
266  size_t length = 0;
267  char *tempLastCommandOutput = NULL;
268  size_t logMessageLength = strlen(logMessage);
269 
270  if (logMessageLength <= 0) {
271  return;
272  }
273 
274  lastCommandOutputLock();
275 
276  if (lastCommandOutput == NULL) {
277  length = logMessageLength + 1;
278 
279  lastCommandOutput = (char*)av_malloc(length);
280  memcpy(lastCommandOutput, logMessage, length);
281  } else {
282  size_t length1 = strlen(lastCommandOutput);
283  length = length1 + logMessageLength + 1;
284 
285  char *newLastCommandOutput = (char*)av_malloc(length);
286  memcpy(newLastCommandOutput, lastCommandOutput, length1);
287  memcpy(newLastCommandOutput + length1, logMessage, logMessageLength + 1);
288 
289  tempLastCommandOutput = lastCommandOutput;
290  lastCommandOutput = newLastCommandOutput;
291  }
292 
293  lastCommandOutputUnlock();
294 
295  if (tempLastCommandOutput != NULL) {
296  av_free(tempLastCommandOutput);
297  }
298 }

300 void monitorWait(int milliSeconds) {
301  struct timeval tp;
302  struct timespec ts;
303  int rc;
304 
305  rc = gettimeofday(&tp, NULL);
306  if (rc) {
307  return;
308  }
309 
310  ts.tv_sec = tp.tv_sec;
311  ts.tv_nsec = tp.tv_usec * 1000;
312  ts.tv_sec += milliSeconds / 1000;
313  ts.tv_nsec += (milliSeconds % 1000)*1000000;
314 
315  pthread_mutex_lock(&monitorMutex);
316  pthread_cond_timedwait(&monitorCondition, &monitorMutex, &ts);
317  pthread_mutex_unlock(&monitorMutex);
318 }

332 void logCallbackDataAdd(int level, const char *data) {
333 
334  // CREATE DATA STRUCT FIRST
335  struct CallbackData *newData = (struct CallbackData*)av_malloc(sizeof(struct CallbackData));
336  newData->type = 1;
337  newData->logLevel = level;
338  size_t dataSize = strlen(data) + 1;
339  newData->logData = (char*)av_malloc(dataSize);
340  memcpy(newData->logData, data, dataSize);
341  newData->next = NULL;
342 
343  mutexLock();
344 
345  // INSERT IT TO THE END OF QUEUE
346  if (callbackDataTail == NULL) {
347  callbackDataTail = newData;
348 
349  if (callbackDataHead != NULL) {
350  LOGE("Dangling callback data head detected. This can cause memory leak.");
351  } else {
352  callbackDataHead = newData;
353  }
354  } else {
355  struct CallbackData *oldTail = callbackDataTail;
356  oldTail->next = newData;
357 
358  callbackDataTail = newData;
359  }
360 
361  mutexUnlock();
362 
363  monitorNotify();
364 }

410 struct CallbackData *callbackDataRemove() {
411  struct CallbackData *currentData;
412 
413  mutexLock();
414 
415  if (callbackDataHead == NULL) {
416  currentData = NULL;
417  } else {
418  currentData = callbackDataHead;
419 
420  struct CallbackData *nextHead = currentData->next;
421  if (nextHead == NULL) {
422  if (callbackDataHead != callbackDataTail) {
423  LOGE("Head and tail callback data pointers do not match for single callback data element. This can cause memory leak.");
424  } else {
425  callbackDataTail = NULL;
426  }
427  callbackDataHead = NULL;
428 
429  } else {
430  callbackDataHead = nextHead;
431  }
432  }
433 
434  mutexUnlock();
435 
436  return currentData;
437 }

447 void mobileffmpeg_log_callback_function(void *ptr, int level, const char* format, va_list vargs) {
448  char line[LOG_LINE_SIZE];
449  AVBPrint part[4];
450  int print_prefix = 1;
451 
452  if (level >= 0) {
453  level &= 0xff;
454  }
455  int activeLogLevel = av_log_get_level();
456  if ((activeLogLevel == AV_LOG_QUIET) || (level > activeLogLevel)) {
457  return;
458  }
459 
460  avutil_log_format_line(ptr, level, format, vargs, part, &print_prefix);
461  avutil_log_sanitize(part[0].str);
462  avutil_log_sanitize(part[1].str);
463  avutil_log_sanitize(part[2].str);
464  avutil_log_sanitize(part[3].str);
465 
466  snprintf(line, sizeof(line), "%s%s%s%s", part[0].str, part[1].str, part[2].str, part[3].str);
467 
468  logCallbackDataAdd(level, line);
469  appendLastCommandOutput(line);
470 
471  av_bprint_finalize(part, NULL);
472  av_bprint_finalize(part+1, NULL);
473  av_bprint_finalize(part+2, NULL);
474  av_bprint_finalize(part+3, NULL);
475 }

495 void *callbackThreadFunction() {
496  JNIEnv *env;
497  jint getEnvRc = (*globalVm)->GetEnv(globalVm, (void**) &env, JNI_VERSION_1_6);
498  if (getEnvRc != JNI_OK) {
499  if (getEnvRc != JNI_EDETACHED) {
500  LOGE("Callback thread failed to GetEnv for class %s with rc %d.\n", configClassName, getEnvRc);
501  return NULL;
502  }
503 
504  if ((*globalVm)->AttachCurrentThread(globalVm, &env, NULL) != 0) {
505  LOGE("Callback thread failed to AttachCurrentThread for class %s.\n", configClassName);
506  return NULL;
507  }
508  }
509 
510  LOGD("Callback thread started.\n");
511 
512  while(redirectionEnabled) {
513 
514  struct CallbackData *callbackData = callbackDataRemove();
515  if (callbackData != NULL) {
516  if (callbackData->type == 1) {
517 
518  // LOG CALLBACK
519 
520  size_t size = strlen(callbackData->logData);
521 
522  jbyteArray byteArray = (jbyteArray) (*env)->NewByteArray(env, size);
523  (*env)->SetByteArrayRegion(env, byteArray, 0, size, (jbyte *)callbackData->logData);
524  (*env)->CallStaticVoidMethod(env, configClass, logMethod, callbackData->logLevel, byteArray);
525  (*env)->DeleteLocalRef(env, byteArray);
526 
527  // CLEAN LOG DATA
528  av_free(callbackData->logData);
529 
530  } else {
531 
532  // STATISTICS CALLBACK
533 
534  (*env)->CallStaticVoidMethod(env, configClass, statisticsMethod,
535  callbackData->statisticsFrameNumber, callbackData->statisticsFps,
536  callbackData->statisticsQuality, callbackData->statisticsSize,
537  callbackData->statisticsTime, callbackData->statisticsBitrate,
538  callbackData->statisticsSpeed);
539 
540  }
541 
542  // CLEAN STRUCT
543  callbackData->next = NULL;
544  av_free(callbackData);
545 
546  } else {
547  monitorWait(100);
548  }
549  }
550 
551  (*globalVm)->DetachCurrentThread(globalVm);
552 
553  LOGD("Callback thread stopped.\n");
554 
555  return NULL;
556 }

565 jint JNI_OnLoad(JavaVM *vm, void *reserved) {
566  JNIEnv *env;
567  if ((*vm)->GetEnv(vm, (void**)(&env), JNI_VERSION_1_6) != JNI_OK) {
568  LOGE("OnLoad failed to GetEnv for class %s.\n", configClassName);
569  return JNI_FALSE;
570  }
571 
572  jclass localConfigClass = (*env)->FindClass(env, configClassName);
573  if (localConfigClass == NULL) {
574  LOGE("OnLoad failed to FindClass %s.\n", configClassName);
575  return JNI_FALSE;
576  }
577 
578  if ((*env)->RegisterNatives(env, localConfigClass, configMethods, 11) < 0) {
579  LOGE("OnLoad failed to RegisterNatives for class %s.\n", configClassName);
580  return JNI_FALSE;
581  }
582 
583  (*env)->GetJavaVM(env, &globalVm);
584 
585  logMethod = (*env)->GetStaticMethodID(env, localConfigClass, "log", "(I[B)V");
586  if (logMethod == NULL) {
587  LOGE("OnLoad thread failed to GetMethodID for %s.\n", "log");
588  (*globalVm)->DetachCurrentThread(globalVm);
589  return JNI_FALSE;
590  }
591 
592  statisticsMethod = (*env)->GetStaticMethodID(env, localConfigClass, "statistics", "(IFFJIDD)V");
593  if (logMethod == NULL) {
594  LOGE("OnLoad thread failed to GetMethodID for %s.\n", "statistics");
595  (*globalVm)->DetachCurrentThread(globalVm);
596  return JNI_FALSE;
597  }
598 
599  av_jni_set_java_vm(vm, NULL);
600 
601  configClass = (jclass) ((*env)->NewGlobalRef(env, localConfigClass));
602 
603  redirectionEnabled = 0;
604 
605  callbackDataHead = NULL;
606  callbackDataTail = NULL;
607 
608  mutexInit();
609  monitorInit();
610  logInit();
611 
612  return JNI_VERSION_1_6;
613 }

717 JNIEXPORT jint JNICALL Java_com_arthenica_mobileffmpeg_Config_nativeExecute(JNIEnv *env, jclass object, jobjectArray stringArray) {
718  jstring *tempArray = NULL;
719  int argumentCount = 1;
720  char **argv = NULL;
721 
722  if (stringArray != NULL) {
723  int programArgumentCount = (*env)->GetArrayLength(env, stringArray);
724  argumentCount = programArgumentCount + 1;
725 
726  tempArray = (jstring *) av_malloc(sizeof(jstring) * programArgumentCount);
727  }
728 
729  /* PRESERVE USAGE FORMAT
730  *
731  * ffmpeg <arguments>
732  */
733  argv = (char **)av_malloc(sizeof(char*) * (argumentCount));
734  argv[0] = (char *)av_malloc(sizeof(char) * (strlen(LIB_NAME) + 1));
735  strcpy(argv[0], LIB_NAME);
736 
737  // PREPARE
738  if (stringArray != NULL) {
739  for (int i = 0; i < (argumentCount - 1); i++) {
740  tempArray[i] = (jstring) (*env)->GetObjectArrayElement(env, stringArray, i);
741  if (tempArray[i] != NULL) {
742  argv[i + 1] = (char *) (*env)->GetStringUTFChars(env, tempArray[i], 0);
743  }
744  }
745  }
746 
747  // LAST COMMAND OUTPUT SHOULD BE CLEARED BEFORE STARTING A NEW EXECUTION
748  clearLastCommandOutput();
749 
750  // RUN
751  int retCode = execute(argumentCount, argv);
752 
753  // CLEANUP
754  if (tempArray != NULL) {
755  for (int i = 0; i < (argumentCount - 1); i++) {
756  (*env)->ReleaseStringUTFChars(env, tempArray[i], argv[i + 1]);
757  }
758 
759  av_free(tempArray);
760  }
761  av_free(argv[0]);
762  av_free(argv);
763 
764  return retCode;
765 }
